清单导入导出框架解析

主要功能类

1. CrlListImportExportUtil

  • 作用: 清单导入导出的统一入口工具类

  • 主要功能:

    • 提供清单导出Excel接口：exportToExcel

    • 提供清单导入Excel接口：importFromExcel

    • 获取清单配置信息：getCrlListCfgItems

2. CrlListBillImportHelper

  • 作用: 清单导入的核心处理类
  • 主要功能:
    • 解析Excel文件并提取数据
    • 处理导入前校验
    • 管理多线程导入过程
    • 执行导入后的数据更新

3. CrlListBillExportHelper

  • 作用: 清单导出的核心处理类
  • 主要功能:
    • 生成Excel工作簿
    • 按配置顺序导出各个页签
    • 管理多线程导出过程
    • 设置导出进度和文件命名

基础框架类

4. CrlListTabImportHelper

  • 作用: 清单页签导入的基础工具类
  • 主要功能:
    • 解析表头数据：nhandleColumns
    • 解析表体数据：nhandleBodyData
    • 保存导入数据：nsave

5. CrlListTabExportHelper

  • 作用: 清单页签导出的基础工具类（未在上下文中显示，但根据架构推断存在）
  • 主要功能:
    • 创建Excel表头
    • 设置表体数据
    • 处理导出格式

缓存和辅助类

6. CrlListImportCache / CrlListExportCache

  • 作用: 分别管理导入和导出过程中的缓存数据

7. CrlListHideSheetHelper

  • 作用: 处理Excel中隐藏页签数据（如枚举、F7下拉项数据）

8. CrlListTabImpExpUtil

  • 作用: 提供导入导出的通用工具方法
  • 功能包括: 读取Excel数据、进度管理等

配置管理类

9. CrlListTabInfo

  • 作用: 定义清单页签的操作工具类集合
  • 包含: 基本操作、标准清单库更新、复制、导出、导入工具类

10. CrlListGetTabHelperUtil

  • 作用: 获取各种操作类型的清单页签工具类实例

数据传输类

11. CrlListExcelColumnPO

  • 作用: Excel列的封装对象，包含列信息、数据类型、值范围等

12. CrlListImportSheetInfo

  • 作用: Excel页签信息封装，包含控制键、启用新接口标志等信息

核心流程

  1. 导出流程: CrlListImportExportUtil.exportToExcel → CrlListBillExportHelper → CrlListTabExportHelper
  2. 导入流程: CrlListImportExportUtil.importFromExcel → CrlListBillImportHelper → CrlListTabImportHelper

   /**
    * 创建Sheet
    * 1. 从模板中取隐藏的默认sheet，更名、显示
    * 2. 动态左树的情况：从模板中取页签对应的默认模板，复制、更名
    */
   public XSSFSheet[] createSheets(XSSFWorkbook workbook, CrlListCfgItemPO cfgItemPO) {
      //页签对应的子页面
      String subPageKey = cfgItemPO.getPageKey();
      CrlListTabInfo listTabInfo = CrlListTabInfoInit.getListTabInfo(subPageKey);
      if(null == listTabInfo) {
         return null;
      }
      
      XSSFSheet[] sheets;
      //1. 创建动态左树对应的sheet
      if(hasDynLeftTree()) {
         String tabPageKey = cfgItemPO.getTabPageKey();
         if(!subPageKey.equals(tabPageKey)) {
            listTabInfo = CrlListTabInfoInit.getListTabInfo(tabPageKey);
         }
         String defaultSheetName = Optional.ofNullable(listTabInfo)
               .map(info -> info.getDefaultExportSheetName())
               .orElse(null);
         if(StringUtils.isEmpty(defaultSheetName)) {
            return null;
         }
         //1.1 取模板中默认的sheet
         int defaultSheetIndex = workbook.getSheetIndex(defaultSheetName);
         
         List<CrlListCfgItemPO> dynLeftTrees = getDynLeftTrees();
         sheets = new XSSFSheet[dynLeftTrees.size()];
         
         //1.2 创建动态左树对应的sheet
         for (int i = 0; i < dynLeftTrees.size(); i++) {
            CrlListCfgItemPO dynLeftTree = dynLeftTrees.get(i);
            String sheetName = dynLeftTree.getDisplayName();
            if(sheetName.length() > 32){
               throw new KDBizException(String.format(ResManager.loadKDString("创建的Excel工作空间名称过长，请修改【%s】对应数据", "CrlListTabExportHelper_0", "crlc-crlsta-business"),sheetName));
            }
            sheets[i] = workbook.cloneSheet(defaultSheetIndex, sheetName);
            sheet_leftTree_mappingData.put(sheetName, dynLeftTree);
            //要实现单元格锁定必须开启写保护，单元格设置时需指定是否锁定
            CrlExcelUtil.protectSheet(sheets[i]);
         }
         //1.3 删除默认的sheet
         workbook.removeSheetAt(defaultSheetIndex);
      }
      //2. 创建固定sheet: 没有左树、固定左树的情况
      else {
         String defaultSheetName = listTabInfo.getDefaultExportSheetName();
         if(StringUtils.isEmpty(defaultSheetName)) {
            return null;
         }
         //取模板中默认的sheet
         XSSFSheet defaultSheet = workbook.getSheet(defaultSheetName);
     XSSFSheet defaultSheet = workbook.getSheet(defaultSheetName);
      String sheetName = cfgItemPO.getDisplayName();
      if(sheetName.length() > 32){
         throw new KDBizException(String.format(ResManager.loadKDString("创建的Excel工作空间名称过长，请修改【%s】对应数据", "CrlListTabExportHelper_0", "crlc-crlsta-business"),sheetName));
      }
      int sheetIndex = workbook.getSheetIndex(defaultSheet);
      workbook.setSheetName(sheetIndex, sheetName);
      workbook.setSheetHidden(sheetIndex, false);
      //要实现单元格锁定必须开启写保护，单元格设置时需指定是否锁定
      CrlExcelUtil.protectSheet(defaultSheet);
      sheets = new XSSFSheet[]{defaultSheet};
   }
   
   return sheets;
}

/**
 * 创建表头（列）
 */
public List<CrlListBillEntryColumnPO> createColumns(XSSFSheet sheet) {
   CrlListCfgItemPO leftTree = hasDynLeftTree() ? sheet_leftTree_mappingData.get(sheet.getSheetName()) : null;

   //获取Sheet, 用于下拉列表的创建
   CrlListDropDownListUtil.getOrcreate().setSheet(sheet);

   List<CrlListBillEntryColumnPO> columns = getColumnsData(leftTree);
   if(null == columns || columns.size() == 0) {
      return null;
   }
   
   XSSFRow headRow1 = sheet.getRow(0);
   for (CrlListBillEntryColumnPO columnPO : columns) {
      XSSFCell headRow1Cell = headRow1.getCell(columnPO.getColumnIndex());
      if(null == headRow1Cell) {
         headRow1Cell = headRow1.createCell(columnPO.getColumnIndex());
      }
      //隐藏列
      sheet.setColumnHidden(columnPO.getColumnIndex(), !columnPO.isColumnShow());
      headRow1Cell.setCellValue(columnPO.getColumnName());
      
      List<CrlListBillEntryColumnPO> childs = columnPO.getChilds();
      if(null != childs && childs.size() > 0) {
         XSSFRow headRow2 = sheet.getRow(1);
         for (CrlListBillEntryColumnPO childPO : childs) {
            XSSFCell headRow2Cell = headRow2.getCell(childPO.getColumnIndex());
            if(null == headRow2Cell) {
               headRow2Cell = headRow2.createCell(childPO.getColumnIndex());
            }
            //隐藏列
            sheet.setColumnHidden(childPO.getColumnIndex(), !childPO.isColumnShow());
            headRow2Cell.setCellValue(childPO.getColumnName());
         }
      }
   }
   return columns;
}

/**
 * 设置表格数据（行）
 */
public void setBodyData(XSSFSheet sheet, List<CrlListBillEntryColumnPO> columns) {
   if(null == columns || columns.size() == 0) {
      return;
   }
   
   CrlListCfgItemPO leftTree = hasDynLeftTree() ? sheet_leftTree_mappingData.get(sheet.getSheetName()) : null;
   List<List<CrlListBillEntryCellPO>> bodyData = getRowsData(leftTree, columns);
   if(null == bodyData || bodyData.size() == 0) {
      return;
   }
   
   int dataStartRowIdx = DATAROW_START_INDEX;
   for (List<CrlListBillEntryCellPO> rowData : bodyData) {
      //如果不是第一行数据，则复制第一行格式
      if(dataStartRowIdx != DATAROW_START_INDEX) {
         sheet.copyRows(DATAROW_START_INDEX, DATAROW_START_INDEX, dataStartRowIdx, cellCopyPolicy);
      }
      XSSFRow row = sheet.getRow(dataStartRowIdx);
      if (row == null){
         row = sheet.createRow(dataStartRowIdx);
      }
      //设置行数据
      setRowData(row, rowData);
      //设置行样式
      setRowStyle(row, rowData);

      dataStartRowIdx++;
   }
   //设置单元格数字校验
   setCellNumberValidation(sheet,columns);
   //设置列宽自适应
   autoColumnWidth(sheet);
}

	public static void autoColumnWidth(XSSFSheet sheet) {
		int maxColumn = sheet.getRow(1).getPhysicalNumberOfCells();
		for(int columnNum = 0; columnNum <= maxColumn; columnNum++) {
			int columnWidth = sheet.getColumnWidth(columnNum) / 256;
			//记录初始宽度
			for(int rowNUm = 0; rowNUm <= sheet.getLastRowNum(); rowNUm++) {
				XSSFRow currentRow;
				if(sheet.getRow(rowNUm) == null) {
					currentRow = sheet.createRow(rowNUm);
				} else {
					currentRow = sheet.getRow(rowNUm);
				}
				if(currentRow.getCell(columnNum) != null) {
					XSSFCell cellPO = currentRow.getCell(columnNum);
					if(cellPO.getCellType() == CellType.STRING) {
						int length = (cellPO.getStringCellValue().getBytes(StandardCharsets.UTF_8).length + cellPO.toString().length()) / 2;
						if(columnWidth < length) {
							columnWidth = length;
						}
					}
				}
			}
			//限制最大宽度不大于15000, 限制最小宽度
			if(columnWidth > 58) {
				sheet.setColumnWidth(columnNum, 58 * 256);
			}
			else {
				sheet.setColumnWidth(columnNum, (3 + columnWidth) * 256);
			}
		}
		for(int rowNUm = 0; rowNUm <= sheet.getLastRowNum(); rowNUm++) {
			short cellHeight = 1;
			XSSFRow currentRow;
			if(sheet.getRow(rowNUm) == null) {
				currentRow = sheet.createRow(rowNUm);
			} else {
				currentRow = sheet.getRow(rowNUm);
			}
			for(int columnNum = 0; columnNum <= maxColumn; columnNum++) {
				if(currentRow.getCell(columnNum) != null) {
					XSSFCell cellPO = currentRow.getCell(columnNum);
					if(cellPO.getCellType() == CellType.STRING) {
						String stringCellValue = cellPO.getStringCellValue();
						short height = getChildStringCount(stringCellValue,sheet.getColumnWidth(columnNum));

						if(cellHeight < height){
							cellHeight = (short) height;
						}
					}
				}
			}
			//行高16.5，1pt=20wints
			currentRow.setHeight((short) (cellHeight * 16 *20));
		}

private static short getChildStringCount(String str,int width) {
		//计数器
		short counter=0;
		//从字符串起始位置开始查找
		int i=0;
		String str_son = "\n";
		//开始查找
		//有换行
		if(str.contains(str_son)) {
			while (str.indexOf(str_son, i) != -1) {   //不等于-1意味着查找到了子串
				int index = str.indexOf(str_son, i);
				String substring = str.substring(i, index);
				counter += Math.ceil(((substring.getBytes(StandardCharsets.UTF_8).length + substring.length()) / 2) / (width / 256.0));
				//从查到的子串结束后开始继续查找（indexOf返回的是查找到的子串的起始位置）
				i = index + str_son.length();
			}
			counter++;
		}else{
			counter = (short) Math.ceil(((str.getBytes(StandardCharsets.UTF_8).length + str.length()) / 2) / (width / 256.0));
		}
		return counter;
	}

	public String exportExcel(DynamicObject listBillObj,
			InputStream defExportTplStream,
			List<CrlListCfgItemPO> subCfgItems,
			CrlListExportTypeEnum exportType) throws
			IOException {
		if(null == listBillObj || null == defExportTplStream || null == subCfgItems || subCfgItems.size() == 0) {
			return null;
		}
		//清单ID
		long listBillId = (long) listBillObj.getPkValue();
		//清单名称
		String listBillName = listBillObj.getString(CrlListBillTplConst.BILLNAME);
		//默认导出模板
		workbook = new XSSFWorkbook(defExportTplStream);
		//初始化导出上下文
		CrlListBillExportContext.getOrCreate().setWorkbook(workbook);
		
		//初始化
		init();
		
		//生成第1个sheet
		XSSFSheet firstSheet = create1stSheet(listBillObj);
		
		int seq = 0;
		for (CrlListCfgItemPO cfgItemPO : subCfgItems) {
			//页签对应的子页面
			String subPageKey = cfgItemPO.getPageKey();
//页签导出工具类
			CrlListTabExportHelper exportHelper = getExportHelper(subPageKey, listBillId, exportType);
			if(null == exportHelper) {
				continue;
			}
			//为ExportHelper增加变量
			addListCustomParams(listBillObj ,exportHelper);
			//1. 创建Sheet
			XSSFSheet[] sheets = exportHelper.createSheets(workbook, cfgItemPO);
			if(null == sheets || sheets.length == 0) {
				continue;
			}

			
			for (XSSFSheet sheet : sheets) {
				//2. 创建表头（列）
				List<CrlListBillEntryColumnPO> columns = exportHelper.createColumns(sheet);
				
				//3. 设置表格数据（行）
				exportHelper.setBodyData(sheet, columns);
				
				//4. 设置第1个sheet中的内容：目录
				set1stSheetData(firstSheet, cfgItemPO, sheet, seq++);

				workbook.setSheetOrder(sheet.getSheetName(),seq+1);
			}
		}
		
		String listFileName = String.format("%s_%s.xlsx", listBillName, CrlDateUtil.formatDate(new Date(), "yyyyMMdd"));
		return CrlbasExportUtil.getExcelDownloadUrl(listFileName, workbook);
	}

    /**
     * 处理表头数据（列）
     */
    public List<CrlListExcelColumnPO> nhandleColumns(Map<Integer, Map<Integer, String>> sheetData, CrlListCfgItemPO leftTreeCfg) {
        // 性能耗时log
        StopWatch started = StopWatch.createStarted();
        logger.info("CrlListTabImportHelper.nhandleColumns start... leftTreeCfg : [{}]", leftTreeCfg.getDisplayName());

        //标识信息
        Map<String, CrlListExcelColumnPO> tempColumns = new LinkedHashMap<>();

        //查看界面字段集合
        Map<String, IDataEntityProperty> vAllFields = EntityMetadataCache.getDataEntityType(leftTreeCfg.getLeftTreePageKey())
                .getAllFields();
        //后台实体字段集合
        Map<String, IDataEntityProperty> dAllFields = Optional.ofNullable(getListTabDataEntityId())
                .map(n -> EntityMetadataCache.getDataEntityType(n).getAllFields())
                .orElse(Collections.emptyMap());

        //标识行
        Map<Integer, String> keyRow = sheetData.get(getKeyHeadRowIndex());
        for (Integer columnIndex : keyRow.keySet()) {
            String vkey = keyRow.get(columnIndex);
            if(tempColumns.containsKey(vkey) || IGNORE_COLUMNS_4IMPORT.contains(vkey)) {
                continue;
            }

            String columnType = null;
            //处理ID字段
            if (vkey.endsWith(".id")) {
                vkey = vkey.substring(0, vkey.length() - 3);
                columnType = CrlListExcelColumnPO.CT_LONG;
            }

            CrlListExcelColumnPO column = new CrlListExcelColumnPO(columnIndex, vkey, null, columnType);

            //处理字段
            IDataEntityProperty vProperty = vAllFields.get(vkey);
            column = nhandleColumn(column, vProperty, dAllFields);

            if (null != column) {
                tempColumns.put(vkey, column);
            }
        }

        // 性能耗时log
        started.stop();
        logger.info("CrlListTabImportHelper.nhandleColumns start... leftTreeCfg : [{}] , result size : [{}], cost: [{}] ms",
                leftTreeCfg.getDisplayName(), tempColumns.size(), started.getTime());

        return new LinkedList<>(tempColumns.values());
    }

    /**
     * 处理字段，忽略的字段不处理
     */
    protected CrlListExcelColumnPO nhandleColumn(CrlListExcelColumnPO column,
                                                 IDataEntityProperty vProperty,
                                                 Map<String, IDataEntityProperty> dAllFields) {
        String columnKey = column.getColumnKey();
        //忽略字段不用处理
        if (!"id".equals(columnKey)
                && !"pid".equals(columnKey)
                && IGNORE_COLUMNS_4IMPORT.contains(columnKey)) {
            return null;
        }

        String columnType = null;
        int scale = -1;
        Object valueRange = null;
        String refBasedataId = null;

        if (vProperty instanceof ComboProp) {
            columnType = CrlListExcelColumnPO.CT_ENUM;

            Map<String, String> values = new HashMap<>();
            List<ValueMapItem> comboItems = ((ComboProp) vProperty).getComboItems();
            for (ValueMapItem item : comboItems) {
                values.put(item.getName().getLocaleValue(), item.getValue());
            }
            valueRange = values;
        } else if (vProperty instanceof TextProp) {
            columnType = CrlListExcelColumnPO.CT_STRING;
        } else if (vProperty instanceof DateTimeProp) {
            columnType = CrlListExcelColumnPO.CT_DATE;
        } else if (vProperty instanceof BigIntProp) {
            columnType = CrlListExcelColumnPO.CT_LONG;
        } else if (vProperty instanceof DecimalProp) {
            columnType = CrlListExcelColumnPO.CT_DECIMAL;
            scale = ((DecimalProp) vProperty).getScale();
        } else if (vProperty instanceof BasedataProp) {
            columnType = CrlListExcelColumnPO.CT_BASEDATA;
            refBasedataId = ((BasedataProp) vProperty).getBaseEntityId();
        } else if( vProperty instanceof BooleanProp) {
            columnType = CrlListExcelColumnPO.CT_CHECKBOX;
        }
        // baseid 特殊处理， 有些页面不加baseid,但是导入解析需要，找不到对应的元数据类型
        if(CrlListTabListTplConst.VIEW_ID.equals(columnKey)
                || CrlListTabListTplConst.VIEW_PID.equals(columnKey)
                || columnKey.contains("baseid")
                || columnKey.contains(String.format("%s._id", CrlListTabListTplConst.VIEW_STDLISTLIB))
                || columnKey.contains(String.format("%s._id",  CrlListDivWorkViewConst.VIEW_MATERIAL))) {
            columnType = CrlListExcelColumnPO.CT_LONG;
        } else if(CrlListTabListTplConst.VIEW_SEQ.equals(columnKey)) {
            columnType = CrlListExcelColumnPO.CT_DECIMAL;
        }

        //通过查看字段找到后台字段
        String propKey = null;
        if (!IGNORE_COLUMNS_4SAVE.contains(columnKey) && dAllFields.size() > 0) {
            IDataEntityProperty dProperty = null;
            String prop = columnKey;
            int i = 0;
            while (i > -1) {
                prop = prop.substring(i);
                dProperty = dAllFields.get(prop);
                if (null != dProperty) {
                    break;
                } else {
                    i = prop.indexOf('_');
                    if(i > -1) {
                        i = i + 1;
                    }
                }
            }
            if(null != dProperty) {
                propKey = dProperty.getName();
            }
        }

        column.setPropKey(propKey);
        column.setColumnType(columnType);
        column.setScale(scale);
        column.setValueRange(valueRange);
        column.setRefBasedataId(refBasedataId);

        return column;
    }

    /**
     * 处理表格数据（行）
     */
    public List<Map<String, Object>> nhandleBodyData(Map<Integer, Map<Integer, String>> sheetData,
                                                     CrlListCfgItemPO leftTreeCfg,
                                                     List<CrlListExcelColumnPO> detailColumns) {
        // 性能耗时log
        StopWatch started = StopWatch.createStarted();
        logger.info("CrlListTabImportHelper.nhandleBodyData start... leftTreeCfg : [{}]", leftTreeCfg.getDisplayName());

        List<Map<String, Object>> rowDatas = new LinkedList<>();
        int dataRowStartIdx = getDefaultDataRowStartIndex();
        Map<String, CrlListImportSheetInfo> sheetInfoCache = cache.getSheetInfoCache(leftTreeCfg.getLeftTreePageKey());
        if(sheetInfoCache != null && sheetInfoCache.size() > 0){
            CrlListImportSheetInfo crlListImportSheetInfo = sheetInfoCache.get(leftTreeCfg.getCtrlKey());
            if (crlListImportSheetInfo != null){
                Integer startRowIndex = crlListImportSheetInfo.getStartRowIndex();
                if(startRowIndex > 0) {
                    dataRowStartIdx = startRowIndex;
                }
            }
        }

        for (Integer rowIndex : sheetData.keySet()) {
            if(rowIndex < dataRowStartIdx) {
                continue;
            }
            Map<Integer, String> row = sheetData.get(rowIndex);
            Map<String, Object> rowData = new HashMap<>();

            for (CrlListExcelColumnPO detailColumn : detailColumns) {
                String columnKey = detailColumn.getColumnKey();
                int columnIndex = detailColumn.getColumnIndex();
                Object value = row.get(columnIndex);
//				if(null == value) {
//					continue;
//				}

                //枚举
                if (detailColumn.isEnum()) {
                    if(StringUtils.isNotBlank((String) value) && null != detailColumn.getValueRange()) {
                        @SuppressWarnings("unchecked")
                        Map<String, String> valueRanges = (Map<String, String>) detailColumn.getValueRange();
                        value = valueRanges.get((String) value);
                    } else if(null == value) {
                        value = "";
                    }
                }
                //F7
                else if (detailColumn.isBaseData()) {
                    if(StringUtils.isBlank((String) value)) {
                        value = null;
                    } else if(StringUtils.isNotBlank(detailColumn.getRefBasedataId())) {
                        Map<String, Object> dropDownList = listHideSheetHelper.getDropDownList(detailColumn.getRefBasedataId());
                        if (null != dropDownList && dropDownList.containsKey(value.toString())) {
                            value = dropDownList.get(value.toString());
                        } else {
                            try {
                                //去掉长整型后面添加的L
                                String str = value.toString();
                                if('L' == str.charAt(str.length() - 1)) {
                                    str = str.substring(0, str.length() - 1);
                                }
                                value = Long.parseLong(str);
                            } catch (NumberFormatException e) {
                                //do-nothing
                            }
                        }
                    } else {
                        value = Long.parseLong(value.toString());
                    }
                }
                //ID等
                else if (detailColumn.isLong()) {
                    if(StringUtils.isNotBlank((String) value)) {
                        try {
                            //去掉长整型后面添加的L
                            String str = value.toString();
                            if('L' == str.charAt(str.length() - 1)) {
                                str = str.substring(0, str.length() - 1);
                            }
                            value = Long.parseLong(str);
                        } catch (NumberFormatException e) {
                            //do-nothing
                        }
                    } else {
                        value = 0L;
                    }
                }
                //小数：处理精度
                else if(detailColumn.isDecimal()) {
                    //空值
                    if(StringUtils.isBlank((String) value)) {
                        //处理空值
                        value = detailColumn.isZeroShowFlag() ? null : 0;
                    } else if(detailColumn.getScale() > 0) {
                        value = CrlDigitalUtil.toBigDecimal(value, detailColumn.getScale());
                    }
                } else if (detailColumn.isInt()) {
                    value = Double.valueOf((String) value).intValue();
                } else if (detailColumn.isCheckBox()) {
                    value = CrlBooleanUtil.toBooleanObject((String) value);
                } else if(detailColumn.isDate()) {
                    try {
                        value = CrlDateUtil.parseDate((String) value, CrlDateUtil.DATE_FMT);
                    } catch (ParseException e) {
                        throw new RuntimeException(e);
                    }
                }

                rowData.put(columnKey, value);
            }

            //处理行数据
            rowData = nhandleRowData(detailColumns, rowDatas, row, rowData);

            if(null != rowData) {
                rowDatas.add(rowData);
            }
        }

        // 性能耗时log
        started.stop();
        logger.info("CrlListTabImportHelper.nhandleBodyData end... leftTreeCfg : [{}] , result size : [{}], cost: [{}] ms",
                leftTreeCfg.getDisplayName(), rowDatas.size(), started.getTime());

        return rowDatas;
    }

    /**
     * 保存
     */
    public void nsave(CrlListCfgItemPO leftTreeCfg, List<CrlListExcelColumnPO> detailColumns, List<Map<String, Object>> datas) {
        // 性能耗时log
        StopWatch started = StopWatch.createStarted();
        logger.info("CrlListTabImportHelper.nsave start... leftTreeCfg : [{}]", leftTreeCfg.getDisplayName());

        if(null == detailColumns || detailColumns.size() == 0) {
            return;
        }
        if(null == datas || datas.size() == 0) {
            return;
        }
        List<Object> allRowsDataPks = ngetAllRowDatasPks(leftTreeCfg);
        // 零星清单 数据为空，这时候excel有新增也要导入
        if((null == allRowsDataPks || allRowsDataPks.size() == 0) &&
                !CrlListSporadicConst.ENTITY_NAME.equals(getListTabDataEntityId())) {
            return;
        }
        //导入的数据（Excel数据）
        Map<Object, Map<String, Object>> datasMap = new LinkedHashMap<>(datas.size());
        int seq = 1;
        for (Map<String, Object> data : datas) {
            data.put(CrlListTabListTplConst.VIEW_SEQ, seq++);
            datasMap.put(data.get("id"), data);
        }

        //导入时，需要更新的字段
        String importSelectProperties = getImportSelectProperties();
        List<DynamicObject> mainObjs = new LinkedList<>();
        List<DynamicObject> dynObjs1 = new LinkedList<>();
        List<DynamicObject> dynObjs2 = new LinkedList<>();

        ///////////////////////////////////////////////////////////////////////
        //分批处理，避免内存溢出
        List<Object> batchPks = new LinkedList<>();
        int allSize = allRowsDataPks.size();
        for (int i = 0; i < allSize; i++) {
            batchPks.add(allRowsDataPks.get(i));
            //分页处理，注意最后一页
            if(batchPks.size() < CrlListBillBatchOpHelper.BATCH_SIZE && i < (allSize - 1)) {
                continue;
            }

            ///////////////////////////////////////////////////////////////////////
            //查数据库已存在数据，更新现有数据
            Collection<?> rowObjs = queryRowDatas(batchPks, importSelectProperties);
            batchPks.clear();

            for (Object rowObj : rowObjs) {
                //后台主对象
                DynamicObject mainObj = null;
                //分录1，如工程量
                Object dynObj1 = null;
                //分录2，如单价
                Object dynObj2 = null;

                if(rowObj instanceof DynamicObject) {
                    mainObj = (DynamicObject) rowObj;
                } else if (rowObj instanceof Tuple) {
                    Tuple<?, ?> tuple = (Tuple<?, ?>) rowObj;
                    mainObj = (DynamicObject) tuple.item1;
                    dynObj1 = tuple.item2;
                    if (rowObj instanceof ThreeTuple) {
                        dynObj2 = ((ThreeTuple<?, ?, ?>) rowObj).item3;
                    }
                }
                if(null == mainObj) {
                    continue;
                }

                Object pk = mainObj.getPkValue();
                //新增式导入：回标清单
                if (!isUpdateImport) {
                    pk = mainObj.get(CrlListTabDTplConst.COPYFROMID);
                }

                //excel行数据
                Map<String, Object> rowData = datasMap.remove(pk);
                if(null == rowData || rowData.size() == 0) {
                    continue;
                }

                //处理单元格数据
                for (CrlListExcelColumnPO detailColumn : detailColumns) {
                    String columnKey = detailColumn.getColumnKey();
                    String propKey = detailColumn.getPropKey();

                    //不用处理的字段
                    if (null == propKey
                            || "id".equals(columnKey)
                            || "pid".equals(columnKey)
                            || IGNORE_COLUMNS_4SAVE.contains(propKey)) {
                        continue;
                    }

                    //处理单元格的值
                    nhandleCellValue(detailColumn, rowData, mainObj, dynObj1, dynObj2);
                }

                rowData.clear();
            }
            DynamicObject[] mainResult = null;
            //保存
            if(rowObjs.size() > 0) {
                Object o = rowObjs.stream().findFirst().get();
                if(o instanceof DynamicObject) {
                    mainResult = rowObjs.stream()
                            .map(obj -> (DynamicObject) obj)
                            .toArray(DynamicObject[]::new);
                    SaveServiceHelper.save(mainResult);
                } else if (o instanceof Tuple) {
                    mainObjs.clear();
                    dynObjs1.clear();
                    dynObjs2.clear();

                    rowObjs.forEach(r -> {
                        Tuple<?, ?> tuple = (Tuple<?, ?>) r;
                        mainObjs.add((DynamicObject) tuple.item1);
                        if(tuple.item2 instanceof Map) {
                            //noinspection unchecked
                            dynObjs1.addAll(((Map<Object, DynamicObject>) tuple.item2).values());
                        }
                        if (r instanceof ThreeTuple) {
                            ThreeTuple<?, ?, ?> tuple1 = (ThreeTuple<?, ?, ?>) r;
                            if(tuple1.item3 instanceof Map) {
                                //noinspection unchecked
                                dynObjs2.addAll(((Map<Object, DynamicObject>) tuple1.item3).values());
                            }
                        }
                    });
                    if(mainObjs.size() > 0) {
                        mainResult = mainObjs.toArray(new DynamicObject[0]);
                        SaveServiceHelper.save(mainResult);
                        mainObjs.clear();
                    }
                    if(dynObjs1.size() > 0) {
                        SaveServiceHelper.save(dynObjs1.toArray(new DynamicObject[0]));
                        dynObjs1.clear();
                    }
                    if(dynObjs2.size() > 0) {
                        SaveServiceHelper.save(dynObjs2.toArray(new DynamicObject[0]));
                        dynObjs2.clear();
                    }
                }
            }

            //清空对象，释放内存
            rowObjs.clear();
        }

        //保存新增的数据， 如零星清单
        nsaveNews(leftTreeCfg, detailColumns, datasMap);

        //清空对象，释放内存
        //通过Collections.singletonList(0L)创建的对象，调用clear()会报错，此处忽略这种情况
        if(allRowsDataPks.size() > 1) {
            allRowsDataPks.clear();
        }
        datasMap.clear();
        datas.clear();

        // 性能耗时log
        started.stop();
        logger.info("CrlListTabImportHelper.nhandleBodyData start... leftTreeCfg : [{}] , allSize size : [{}], cost: [{}] ms",
                leftTreeCfg.getDisplayName(), allSize, started.getTime());
    }

    /**
     * 处理单元格的值
     */
    protected void nhandleCellValue(CrlListExcelColumnPO detailColumn,
                                    Map<String, Object> rowData,
                                    DynamicObject mainObj,
                                    Object... dynObjs) {
        String propKey = detailColumn.getPropKey();
        if(!mainObj.containsProperty(propKey)) {
            return;
        }

        String columnKey = detailColumn.getColumnKey();
        Object value = rowData.get(columnKey);

        mainObj.set(propKey, value);
    }

    /**
     * 获得表体数据的所有主键，用于分批查询（当数据过大时，避免内存溢出）
     */
    protected List<Object> ngetAllRowDatasPks(CrlListCfgItemPO leftTreeCfg) {
        return QueryServiceHelper.queryPrimaryKeys(getListTabDataEntityId(),
                new QFilter[]{
                        new QFilter(CrlListTabDTplConst.BILLLISTID, QFilter.equals, listBillId)
                },
                CrlListTabDTplConst.ENTRYSEQ,
                -1
        );
    }

    /**
     * 查询数据，按pks分批查询
     */
    protected Collection<?> queryRowDatas(List<Object> pks, String importSelectProperties) {
        return Optional.ofNullable(getListTabDataEntityId())
                .map(k -> BusinessDataServiceHelper.loadFromCache(k,
                                importSelectProperties,
                                new QFilter[]{
                                        new QFilter(CrlListTabDTplConst.BILLLISTID, QFilter.equals, listBillId),
                                        new QFilter(CrlListTabDTplConst.ID, QFilter.in, pks)
                                },
                                CrlListTabDTplConst.ENTRYSEQ
                        ).values()
                )
                .orElse(null);
    }

    /**
     * 从Excel导入清单数据
     *
     * @param subCfgItems 清单配制信息
     * @param callBack
     */
    public void importExcel(List<CrlListCfgItemPO> subCfgItems, CrlListTabHelperCallBack callBack) {
        StopWatch stopWatch = StopWatch.createStarted();

        if (null == subCfgItems || subCfgItems.size() == 0) {
            return;
        }

        //设置初始进度条
        updateExportProcess(0.03);

        try {
            //标记是否导入保存
            cache.put("isImportForSave", null == callBack);

            //目录页签
//            Map<Integer, Map<Integer, String>> contentSheet = workbookData.get(CrlListBillExportHelper.LIST_CONTENT_SHEET_NAME);
            Map<String, Object> listBillInfoMap = getListBillInfo();

            ///////////////////////////////////////////////////////////////////////
            //导入前校验
            logger.info(String.format("[List Import][%s]: checkBeforeImport", listBillName));
            checkBeforeImport(listBillObj, listBillInfoMap);

            ///////////////////////////////////////////////////////////////////////
            //导入类型：是否是更新式导入
//          boolean isUpdateImport = isUpdateImport(listBillObj, contentSheet);
            boolean isUpdateImport = listBillObj.getPkValue().equals(listBillInfoMap.get(IMPORT_LISTBILL_ID));

            //页签信息
            logger.info(String.format("[List Import][%s]: getListTabSheets", listBillName));
//            Map<String, List<ThreeTuple<String, Boolean, String>>> sheetInfos = CrlListTabImpExpUtil.getListTabSheetInfos(contentSheet);
            Map<String, Map<String, CrlListImportSheetInfo>> sheetInfos = getListTabInfos();
            cache.setSheetInfoCache(sheetInfos);

            //启用了综合单价分析表，并且没有导出分部分项表，才需要同步综合单价分析表的数据到分部分项
            boolean listCompUPriceFlag = sheetInfos.containsKey(CrlListCompUPriceViewConst.ENTITY_NAME)
                    && !sheetInfos.containsKey(CrlListDivWorkViewConst.ENTITY_NAME);
            //是否导入了费用汇总表
            boolean listCostSumFlag = sheetInfos.containsKey(CrlListCostSumViewConst.ENTITY_NAME);

            int allSheetSize = 0;

            //先创建导入工具类，存在动态页签的只需要创建一次，执行一次初始化
            for (CrlListCfgItemPO cfgItemPO : subCfgItems) {
                //页签对应的子页面
                String leftTreePageKey = cfgItemPO.getLeftTreePageKey();

                Map<String, CrlListImportSheetInfo> targetSheetInfos = sheetInfos.get(leftTreePageKey);
                if(null == targetSheetInfos || targetSheetInfos.size() == 0) {
                    continue;
                }
                //页签导出工具类
                CrlListTabImportHelper importHelper = getImportHelper(leftTreePageKey, isUpdateImport);
                if (null == importHelper) {
                    //删除不用导入的sheet数据
                    for (Map.Entry<String, CrlListImportSheetInfo> targetSheet : targetSheetInfos.entrySet()) {
                        workbookData.remove(targetSheet.getValue().getListTabName());
                    }
                    continue;
                }
                allSheetSize += targetSheetInfos.size();
            }
            int i = 0;
            for (CrlListCfgItemPO cfgItemPO : subCfgItems) {
                //页签对应的子页面
                String leftTreePageKey = cfgItemPO.getLeftTreePageKey();
                /*
                 * item1: leftTreeKey,
                 * item2: enableNewInterfaceFlag,
                 * item3: listTabName
                 */
                Map<String, CrlListImportSheetInfo> targetSheets = sheetInfos.get(leftTreePageKey);
                if(null == targetSheets || targetSheets.size() == 0) {
                    continue;
                }
                //页签导出工具类 只是看有没有，没有实际作用
                CrlListTabImportHelper importHelperDemo = getImportHelper(leftTreePageKey, isUpdateImport);
                if (null == importHelperDemo) {
                    continue;
                }

                int sheetSize = targetSheets.size();
                List<CompletableFuture> completableFutureList = new ArrayList<>();
                for (String ctrlKey : targetSheets.keySet()) {
                    CrlListImportSheetInfo sheetInfo = targetSheets.get(ctrlKey);
                    i++;
                    // 获得新的导入类实例，否则导入类成员变量structid 会覆盖
                    CrlListTabImportHelper importHelper = getImportHelper(leftTreePageKey, isUpdateImport);
                    completableFutureList.add(CompletableFuture.runAsync(() -> {
                        String sheetName = sheetInfo.getListTabName();
                        logger.info(String.format("[List Import][%s]: %s", listBillName, sheetName));
                        Map<Integer, Map<Integer, String>> sheetData = workbookData.remove(sheetName);
                        if (null == sheetData || sheetData.size() == 0) {
                            return;
                        }

                        //0. 重置配制信息
                        CrlListCfgItemPO leftTreeCfg = cfgItemPO;
                        //存在动态左树的情况，设置左树ID
                        if (sheetSize > 1) {
                            leftTreeCfg = cfgItemPO.clone();
                        }
                        leftTreeCfg.setCtrlKey(sheetInfo.getCtrlKey());
                        leftTreeCfg.setDisplayName(sheetName);
                        cache.putSheetCfg(sheetName, leftTreeCfg);


                        //清单V2.0的导入接口（建发）
                        if (sheetInfo.getEnableNewInterfaceFlag()) {
                            //1. 处理表头数据（列）
                            List<CrlListExcelColumnPO> detailColumns = importHelper.nhandleColumns(sheetData, leftTreeCfg);

                            //2. 处理表格数据（行）
                            List<Map<String, Object>> datas = importHelper.nhandleBodyData(sheetData, leftTreeCfg, detailColumns);

                            //3. 如果存在回调函数，则调用回调函数处理，否则调用保存方法
                            if (null != callBack) {
                                Map<String, Object> params = new HashMap<>();
                                params.put("key", leftTreeCfg);
                                params.put("title", detailColumns);
                                params.put("value", datas);
                                callBack.callback(params);
                                params.clear();
                            } else {
                                importHelper.nsave(leftTreeCfg, detailColumns, datas);
                            }

                            detailColumns.clear();
                            datas.clear();
                        }
                        //清单V1.0的导入接口（保利）
                        else {
                            //1. 处理表头数据（列）
                            List<CrlListBillEntryColumnPO> columns = importHelper.handleColumns(sheetData, leftTreeCfg);
                            if (null == columns || columns.size() == 0) {
                                return;
                            }

                            //2. 处理表格数据（行）
                            List<Map<String, Object>> datas = importHelper.handleBodyData(sheetData, columns);
                            if (null == datas || datas.size() == 0) {
                                return;
                            }

                            //3. 如果存在回调函数，则调用回调函数处理，否则调用保存方法
                            if (null != callBack) {
                                Map<String, Object> params = new HashMap<>();
                                params.put("key", leftTreeCfg);
                                params.put("title", columns);
                                params.put("value", datas);
                                callBack.callback(params);
                                params.clear();
                            } else {
                                importHelper.save(columns, datas);
                            }

                            columns.clear();
                            datas.clear();
                        }
                    }, executorService));


                    //设置进度条
                    double progress = 1.0 * i / allSheetSize;
                    updateExportProcess(Math.min(progress, 0.5));
                }
                // 等待所有的执行完成
                CompletableFuture.allOf(completableFutureList.toArray(new CompletableFuture[0])).get();
                updateExportProcess(0.95);

                afterImportCfgItem(cfgItemPO);
            }

            //导入后置处理
            logger.info(String.format("[List Import][%s]: afterImportExcel", listBillName));
            afterImportExcel(listBillObj, listCompUPriceFlag, listCostSumFlag);
        } catch (ExecutionException | InterruptedException e) {
            logger.error(e.getMessage(), e);
            throw new RuntimeException(e);
        } finally {
            stopWatch.stop();
            //进度条：完成
            updateExportProcess(1);
            logger.info("[List Import] end listBillName: [{}]:  listBillId: [{}], cost: [{}] ms ",
                    listBillName, listBillId, stopWatch.getTime());

            workbookData.clear();

            //导入完成后，清空缓存
            cache.clear();
        }
    }
    private void afterImportCfgItem(CrlListCfgItemPO cfgItemPO) {
        if(cfgItemPO.getMainTabPageKey().equals(CrlListDivWorkViewConst.ENTITY_NAME)){
            Stopwatch stopwatch = Stopwatch.createStarted();
            DynamicObject[] structs = CrlListStructHelper.getStructs(listBillId);
            Set<Long> structIds = Arrays.stream(structs).map(struct -> struct.getLong(CrlListStructConst.ID)).collect(Collectors.toSet());
            CrlListPriceHelper.updatePriceAndWorkLoad(listBillId,CrlListDivWorkConst.ENTITY_NAME,null,null, structIds,true);
            logger.info("CrlListDivWorkImportHelper.afterNsave CrlListPriceHelper.updatePriceAndWorkLoad  同时更新单价和工程量耗时{}",stopwatch.stop());
        }
    }

    /**

    protected void checkBeforeImport(DynamicObject listBillObj, Map<String, Object> listBillInfo) {
        String errorMsg = null;

        Long srcBillId = (Long) listBillInfo.get(IMPORT_LISTBILL_ID);
        String excelFileType = (String) listBillInfo.get(IMPORT_LISTBILL_ENTITYNAME);

        String thisFileType = listBillObj.getDynamicObjectType().getName();

        logger.info("[List Import][{}]: checkBeforeImport; srcBillId:{},excelFileType:{},thisFileType:{}", listBillName, srcBillId, excelFileType, thisFileType);

        if (thisFileType.equals(excelFileType)) {
            //本类型清单导入，需要校验listbillid是否相同
            if (!listBillObj.getPkValue().equals(srcBillId)) {
                logger.info("[List Import][{}]: checkBeforeImport; 同清单导入失败，文件不匹配", listBillName);
                errorMsg = ResManager.loadKDString("清单文件不匹配，请重新选择", "CrlListBillImportHelper_0", "crlc-crlsta-business");
            }
        }
        //回标清单上传校验
        else if (CrlListFileTypeEnum.CRLBDL_TENLISTBILL.getKey().equals(thisFileType)
                && CrlListFileTypeEnum.CRLBDL_BIDLISTBILL.getKey().equals(excelFileType)) {
            //关联招标清单
            DynamicObject bidListBill = listBillObj.getDynamicObject("crlbdlproject");
            if(bidListBill == null){
                logger.info("[List Import][{}]: checkBeforeImport; 招标倒回标导入失败，未关联招标清单", listBillName);
                errorMsg = ResManager.loadKDString("当前回标清单未关联招标清单，请重新选择", "CrlListBillImportHelper_1", "crlc-crlsta-business");
            }
            //本类型清单导入，需要校验listbillid是否相同
            else if (bidListBill.getLong(CrlListBillF7Const.SRCLISTBILLID) != srcBillId) {
                logger.info("[List Import][{}]: checkBeforeImport; 招标倒回标导入失败，清单文件不匹配", listBillName);
                errorMsg = ResManager.loadKDString("清单文件不匹配，请重新选择", "CrlListBillImportHelper_0", "crlc-crlsta-business");
            }
        } else {
            errorMsg = ResManager.loadKDString("清单文件不匹配，请重新选择", "CrlListBillImportHelper_0", "crlc-crlsta-business");
        }

        if(null != errorMsg) {
            throw new KDBizException(new ErrorCode(IMPORT_CHECK_CODE, errorMsg));
        }
    }

    /**
     * Excel导入后执行
     *
     * 注：该方法在所有页签全部处理后执行
     * 使用场景：
     * 1.报表数据更新，如费用汇总
     */
    protected void afterImportExcel(DynamicObject listBillObj, boolean listCompUPriceFlag, boolean listCostSumFlag) {
        StopWatch stopWatch = StopWatch.createStarted();

        //清单ID
        long listBillId = (long) listBillObj.getPkValue();
        String listBillName = listBillObj.getDataEntityType().getName();

        CrlMethodUtil.asyncInvokeMethod(() -> {
            // 分布分项最顶级节点不导出，所以导入代码更新不到，手动调用向上汇总- 不能放在分部分项导入,每个线程都会执行,会造成重复执行
            CrlListPriceHelper.updateSumToParent(listBillId, CrlListDivWorkConst.ENTITY_NAME);
        });

        logger.info("CrlListBillImportHelper.afterImportExcel CrlListPriceHelper.updateSumToParent  耗时:{}",stopWatch.getTime());
        stopWatch.stop();
        stopWatch.reset();
        stopWatch.start();

        //TODO 反写数据到分部分项 ？ 是否还需要
        if (listCompUPriceFlag) {
            CrlListCompUPriceHelper.syncData2DivWork(listBillName, listBillId);
        }

        logger.info("CrlListBillImportHelper.afterImportExcel CrlListCompUPriceHelper.syncData2DivWork  耗时:{}",stopWatch.getTime());
        stopWatch.stop();
        stopWatch.reset();
        stopWatch.start();


        //异步更新报表数据：
        CrlMethodUtil.asyncInvokeMethod(() -> {
            updateListRptData(listBillObj);
        });

        logger.info("CrlListBillImportHelper.afterImportExcel updateListRptData  耗时:{}",stopWatch.getTime());
        stopWatch.stop();
        stopWatch.reset();
        stopWatch.start();

        //调用费用汇总保存，因为发现如果只导入了本清单，费用汇总没有自动计算
        //没有导入费用汇总表，才需要重新计算，否则只需要计算总造价
        new CrlListCostSumOpHelper().save(listBillObj);

        logger.info("CrlListBillImportHelper.afterImportExcel CrlListCostSumHelper.updateBidListTotalPrice  耗时:{}",stopWatch.getTime());
        stopWatch.stop();
        stopWatch.reset();
        stopWatch.start();

        stopWatch.stop();
        logger.info("CrlListBillImportHelper.afterImportExcel listBillId:[{}], listBillName:[{}] listCompUPriceFlag: [{}], " +
                        "listCostSumFlag:[{}]",
                listBillId, listBillName, listCompUPriceFlag, listCostSumFlag);

    }

    /**
     * 获得导入的清单信息
     */
    protected Map<String, Object> getListBillInfo() {
        //优先取隐藏页签的数据
        Map<String, Object> listBillInfo = CrlListHideSheetHelper.getListBillInfo(workbookData.get(CrlListHideSheetHelper.LISTBILL_INFOS_SHEET));
        if (listBillInfo.size() > 0) {
            return listBillInfo;
        }

        //隐藏页签不存在时，再取第一个页签的数据
        Map<Integer, Map<Integer, String>> sheetDataMap = workbookData.get(CrlListBillExportHelper.LIST_CONTENT_SHEET_NAME);

        Map<Integer, String> row1 = sheetDataMap.get(0);
        Map<Integer, String> row2 = sheetDataMap.get(1);
        //清单ID
        Object id = row1.get(2);
        if (null == id) {
            id = 0L;
        } else {
            id = Long.valueOf((String) id);
        }
        //清单文件类型
        String listEntityName = row2.get(2);

        listBillInfo.put(IMPORT_LISTBILL_ID, id);
        listBillInfo.put(IMPORT_LISTBILL_ENTITYNAME, listEntityName);

        return listBillInfo;
    }
