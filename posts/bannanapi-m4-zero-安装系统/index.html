添加Armbian仓库

1
2
3

echo "deb [arch=arm64] http://apt.armbian.com $(lsb_release -cs) main" | sudo tee /etc/apt/sources.list.d/armbian.list
sudo apt-key adv --keyserver keyserver.ubuntu.com --recv-keys 9F0E78D1
sudo apt update
